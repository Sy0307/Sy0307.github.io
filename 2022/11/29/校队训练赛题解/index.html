CF/gym104021F-Function!

题解-lintkey

$$f_a^{-1}(b)=log_ab$$

故

$$ans=\sum_{a=2}^n\left(a\sum_{b=a}^n\lfloor log_ab\rfloor\lceil1\rceil\right)$$

其中，对于内层的求和 $a\sum_{b=a}^n\lfloor log_ab\rfloor$ 可以 $O(log(n))$ 求出，因为对于区间 $b\in[a^k, a^{k+1})$，都有 $\lfloor log_ab\rfloor==k$

其次，当 $a>\sqrt n$ 时，$\lfloor log_ab\rfloor==1$ 恒成立，上式可化简为

$$\sum_{a=\lfloor\sqrt n\rfloor+1}^na\times(n-a+1)=\sum_{a=\lfloor\sqrt n\rfloor+1}^na\times(n+1)-a^2$$

结合平方数列求和公式 $n(n+1)(2n+1)/6$ 和等差求和公式，可 $O(1)$ 算出上式结果

综上，总复杂度 $O(\sqrt n\ log(n))$

代码：

#include <bits/stdc++.h>
#define inl inline
#define con const
#define cer constexpr
#define ret return
using namespace std;
using us = unsigned int;
using ul = unsigned long long;

template<class T>
inl T pow(T x, us nth) {
    T res(x!=0||nth==0);
    for(; nth; x*=x, nth>>=1)
        if(nth&1) res *= x;
    ret res;
}

template<class T, T M, class U=ul>
struct MT {
    T raw;
    inl cer MT(con T _raw=T()) : raw(move(_raw)) {} // 同类型不预取模!
    template<class OT>
    inl cer MT(con OT _raw=OT()) : raw(_raw%M) {};
    inl cer operator T() con { ret raw; }
    inl MT& add(con MT &r) { raw+=r.raw; if(raw>=M) raw-=M; ret *this; }
    inl MT& sub(con MT &r) { if(r.raw>raw) raw+=M-r.raw; else raw-=r.raw; ret *this; }
    inl MT& mul(con MT &r) { raw = (U(raw) * r.raw)%M; ret *this; }
    inl MT operator-() { MT r; ret r.raw=M-raw, r; }
    inl MT& operator=(con MT &r) { ret raw=r.raw, *this; }
    inl MT& operator+=(con MT &r) { ret add(r); }
    inl MT& operator-=(con MT &r) { ret sub(r); }
    inl MT& operator*=(con MT &r) { ret mul(r); }
    friend inl MT operator+(MT l, con MT &r) { ret l.add(r); }
    friend inl MT operator-(MT l, con MT &r) { ret l.sub(r); }
    friend inl MT operator*(MT l, con MT &r) { ret l.mul(r); }
};
// <- -- --- ----header end---- --- -- ->
cer us MOD = 998244353;
using ms = MT<us, MOD>;
con ms INV6 = pow(ms(6), MOD-2);
con ms INV2 = pow(ms(2), MOD-2);

ul n;

// 平方数列和
inl ms count(ms x) {
    ret x*(x+ms(1))*(x+x+ms(1))*INV6;
}

inl void solve() {
    us mx = sqrt(n);
    ms ans(0);
    for(us a=2, k; a<=mx; ++a) {
        con ul LIM = n / a; // 防a**2超范围
        // [l, r)区间log_a相等
        ul l = a, r = l*a; k = 1;
        while(r<=n) {
            ans += ms(a) * ms(r-l) * ms(k++);
            r = (l=r)>LIM ? n+1 : l*a;
        }
        ans += ms(a) * ms(r-l) * ms(k);
    }
    ans += count(mx)-count(n);
    ans += ms(1ul+n+mx)*ms(n+1) * ms(n-mx) * INV2;
    cout << ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);
    cin >> n; solve();
    ret 0;
}

CF/gym103470J-Xingqiu’s Joke

题解-lintkey

注意到，前两种操作都不会影响到两数之差，由此可知，第三种操作的除数必须是差的质因数

质因数有限，直接枚举质因数记忆化搜索的规模不会太大

#include <bits/stdc++.h>
#define inl inline
#define con const
#define cer constexpr
#define ret return
#define whi(...) while(__VA_ARGS__)
using namespace std;
using us = unsigned int;
using ul = unsigned long long;

template<class T> inl void tmx(T &lhs, con T &rhs) { if(rhs>lhs) lhs = rhs; }
template<class T> inl void tmn(T &lhs, con T &rhs) { if(rhs<lhs) lhs = rhs; }
// <- -- --- ----header end---- --- -- ->
us pri[114], pct, a, b;
unordered_map<ul, us> dp;

#define com(l, r) ((ul(l)<<32)|r)
us dfs(us x, us d) {
    if(x==1 || d==1) ret x - 1;
    if(dp[com(x, d)]) ret dp[com(x, d)];

    us res = x - 1; // 仅一操作
    for(us i=0; i<pct; ++i)
        if(d%pri[i] == 0) {
            con us tmp = x % pri[i];
            tmn(res, min(
                tmp + 1 + dfs(x/pri[i], d/pri[i]),              // 减对齐到质数
                (pri[i]-tmp) + 1 + dfs(x/pri[i]+1, d/pri[i])    // 加对齐到质数
            ));
        }

    ret dp[com(x, d)] = res;
}

us _t = 1;
inl void solve() {
    cout << dfs(a, b-a);
    if(--_t) cout << '\n';
}

inl void pre() {
    dp.clear(); pct = 0;
    cin >> a >> b;
    if(a>b) swap(a, b);
    us d = b - a;
    con us LIM = sqrt(d);
    for(us i=2; i<=LIM; ++i)
        if(d%i == 0) {
            pri[pct++] = i;
            whi(d%i == 0) d /= i;
        }
    if(d>1) pri[pct++] = d;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);
    cin >> _t; whi(_t) pre(), solve();
    ret 0;
}

Kirill and Company

题解 - Sy03

看到题目里k最多为6，很显然这是一个状压问题。

定义一个dp[i][j] 表示在走到i点的时候，是否可以容纳j状态的人。

毫无疑问，由于涉及最短路，我们需要先跑一次bfs。然后每次bfs的时候，再判断下当前的状态是否能够带别人回家（状态转移）。

最后就是一个类似于背包的转移，这里压缩掉了一维。

时间复杂度是$64^2*n^2$，也就是O(能过)。

顺便说bitset是个好东西，跑起来飞快。

注意一个坑，bfs的时候要记得打标记，不然可能会相互到达这样子。

vector<int> edge[maxn];
int dp[10010][1 << 7];
void solve()
{

    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        memset(dp[i], 0, sizeof(dp[i]));
    fr(i, 0, n)
    {
        edge[i].clear();
    }
    fr(i, 1, n)
    {
        edge[i].clear();
    }
    fr(i, 1, m)
    {
        int a, b;
        cin >> a >> b;
        edge[a].push_back(b);
        edge[b].push_back(a);
    }

    int f;
    cin >> f;
    vector<int> h(f + 1);
    fr(i, 1, f) cin >> h[i];

    int t;
    cin >> t;
    vector<int> p(t + 1);
    fr(i, 1, t)
    {
        cin >> p[i];
    }

    vector<pii> q;
    q.push_back({1, 0});
    vector<int> vis(n + 1, 0);
    dp[1][0] = 1;
    while (q.size())
    {
        vector<pii> nxt;
        for (auto &k : q)
        {
            vis[k.first] = 1;
        }
        for (auto &now : q)
        {
            int u = now.first;
            int st = now.second;
            for (auto v : edge[u])
            {
                if (vis[v])
                    continue;
                int nst = st;
                fr(i, 1, t)
                {
                    if (v == h[p[i]])
                    {
                        nst |= (1 << (i - 1));
                    }
                }
                if (!dp[v][nst] && !vis[v])
                {
                    nxt.push_back({v, nst});
                    dp[v][nst] = +1;
                }
            }
        }
        q = nxt;
    }

    fr(i, 1, t) h[p[i]] = 0;
    bitset<(1 << 7)> kk;
    kk[0] = 1;
    fr(i, 1, f)
    {
        if (h[i] == 0)
            continue;
        int now = h[i];
        auto tmp = kk;
        for (int x = 0; x < 64; x++)
        {
            for (int y = 0; y < 64; y++)
            {
                if (dp[now][y])
                {
                    tmp[x | y] = tmp[x | y] | kk[x];
                }
            }
        }
        kk = tmp;
    }
    int ans = 0;
    fr(i, 0, (1 << t) - 1) if (kk[i])
    {
        ans = max(ans, (ll)__builtin_popcount(i));
    }
    cout << t - ans << endl;
}

Hashing Trees

题解 - Sy03

随便拉的一个简单构造。

可以想见的一个结论是，有且仅有可能是上下两层结点都是多个的情况下，才会有异构树的情况（不perfect）。

然后就是考虑不perfect的情况下，怎么输出了。

两种方案的话，第一种就是所有子结点都放在最右端的那个叶子上，然后构造就行了。

第二种就是，大概还和第一种方法差不多，只要把两层之间的最后一个子结点接到上面一层的倒数第二个结点上就行了。

很简单吧qwq

void solve()
{
    int n;
    cin >> n;
    vector<int> h(n + 1);
    fr(i, 0, n)
    {
        cin >> h[i];
    }
    bool suc = 0;
    fr(i, 1, n)
    {
        if (h[i] != 1 && h[i - 1] != 1)
        {
            suc = 1;
            break;
        }
    }
    if (!suc)
    {
        cout << "perfect" << endl;
        return;
    }
    cout << "ambiguous" << endl;
    cout << 0 << " ";
    int root = 0;
    root += h[0];
    fr(i, 1, n)
    {
        fr(j, 1, h[i])
        {
            cout << root << " ";
        }
        root += h[i];
    }
    cout << endl;
    root = 0;
    cout << 0 << " ";
    root += h[0];
    fr(i, 1, n)
    {
        if (h[i] != 1 && h[i - 1] != 1)
        {
            fr(j, 1, h[i] - 1)
            {
                cout << root << " ";
            }
            cout << root - 1 << " ";
            root += h[i];
        }else
        {
            fr(j, 1, h[i])
            {
                cout << root << " ";
            }
            root += h[i];
        }
    }
}

Two Melodies

题解-Sy03

很容易先想到一个$n^3$的暴力。

接着就是去思考怎么去优化这个暴力，只要把一个$n^2$优化成$n$显然就可以了。

注意到$a_i$的范围很重要，由于最大只有$1e5$，那么可以通过这个来加速。

那么接下来就很显然了，具体的可以看看代码。

其实是个很常见的套路吧qwq，感觉cf的rating tag给高了，毕竟是远古题，我觉得只有1600左右的难度罢。

int dp[maxn][maxn];
int a[maxn], maxmod[10], maxnum[100010];
void solve()
{
    int n;
    cin >> n;
    fr(i, 1, n)
    {
        cin >> a[i];
    }
    memset(dp, -0x3f, sizeof(dp));
    dp[0][0] = 0;
    int ans = 0;
    fr(i, 0, n)
    {
        memset(maxmod, 0, sizeof(maxmod));
        memset(maxnum, 0, sizeof(maxnum));
        fr(j, 1, i)
        {
            maxmod[a[j] % 7] = max(maxmod[a[j] % 7], dp[i][j]);
            maxnum[a[j]] = max(maxnum[a[j]], dp[i][j]);
        }
        fr(j, i + 1, n)
        {
            dp[i][j] = max(dp[i][j], max({maxnum[a[j] - 1], maxnum[a[j] + 1], maxmod[a[j] % 7]}))+1;
            dp[i][j] = max(dp[i][j], dp[i][0] + 1);
            dp[j][i] = dp[i][j];
 
            maxnum[a[j]] = max(maxnum[a[j]], dp[i][j]);
            maxmod[a[j] % 7] = max(maxmod[a[j] % 7], dp[i][j]);
            ans = max(dp[i][j], ans);
        }
    }
    cout << ans << endl;
}

CF 1656D-K-good

题解-cyj

  • 考虑 bi 和 bi−1 的关系。我们先假设 b1∼bi−1 均合法。

    • 若 bi−1=bi，则新增的两个数只需一大一小地取即可，bi 也一定合法。
    • 若 bi−1<bi，因为 bi−1 是 2i−3 个数的中位数，所以序列中一定有 i−1 个数比 bi 小。对于任意一个 bj(j<i−1) ，如果其满足 bj≤bi−1，则它可以作为这 i−1 个数中的一个数，不影响结果。若其满足 bi−1<bj<bi，则此时我们发现序列中至少有 i 个数比 bi 小，这就是不合法的。
    • 若 bi−1>bi，同理寻找有无一个 bj 满足 bi−1>bj>bi 即可。

    我们发现所有相同的数出现一次和出现多次在统计答案时没有区别，因为可以简单地使用一个 set 维护，每次查询时：

    • 若 bi−1<bi，查询 bi−1 的后继是否为 bi，若不是则不存在合法序列 a。
    • 若 bi−1>bi，查询 bi 的后继是否为 bi−1。

代码

#include <bits/stdc++.h>
using namespace std;
const int N = 2e5 + 5;
int T, n;
int b[N];
set<int> s;
int main() {
    scanf("%d", &T);
    while (T--) {
        bool fl = 1;
        s.clear();
        scanf("%d", &n);
        for (int i = 1; i <= n; i++) {
            scanf("%d", &b[i]);
            if (!s.count(b[i]))
                s.insert(b[i]);
            if (b[i] == b[i - 1])
                continue;
            if (b[i - 1] < b[i]) {
                if (s.upper_bound(b[i - 1]) != s.find(b[i])) {
                    fl = 0;
                }
            } else {
                if (s.upper_bound(b[i]) != s.find(b[i - 1])) {
                    fl = 0;
                }
            }
        }
        if (fl) {
            printf("YES\n");
        } else {
            printf("NO\n");
        }
    }
    return 0;
}

CF 1656E-Equal Tree Sums

题解-cyj

考虑钦定哪些 # 为 0，那么剩下的 # 的权值也就确定了，为最近的 0 到它的距离。

为什么确定了 0 的位置整张图就确定了？

假设有一个权值为正的 #，相邻的点的权值集合 S。

$|S|=1$ 时，权值是确定的；
$|S|>1$ 时，
$\max S-\min S\le2$，能够得到 # 确定的值，且只有一个；
否则，无论 # 值为多少，整张图都不满足相邻点权值差不超过 11 的要求。
设 # 个数为 cnt，所以答案为 $2^{cnt}-[cnt=nm]$ ，注意没有 0 的情况。

代码

#include <bits/stdc++.h>
using ll = long long;
const int P = 1e9 + 7;
int main() {
    std::ios::sync_with_stdio(false), std::cin.tie(nullptr);
    int cases;
    std::cin >> cases;
    while (cases--) {
        int n, m;
        std::cin >> n >> m;
        std::vector<std::string> a(n);
        int ans = 1;
        bool zero = false;
        for (int i = 0; i < n; ++i) {
            std::cin >> a[i];
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == '0')
                    zero = true;
                else
                    ans = (ll)ans * 2 % P;
            }
        }
        if (!zero)
            ans = (ans + P - 1) % P;
        std::cout << ans << '\n';
    }

    return 0;
}

CF 1656F-Parametric MST

题解-cyj

最后停止的局面，连续的空格长度一定不超过 1，且去掉空格后的环一定满足相邻两点颜色不同。

由于这是个环，所以可以确定最后的 A,B 数量是相等的，步数是偶数，所以后手必胜。

每个人每时刻任何合法的选择都是最优策略。

考虑枚举最后空格的个数、空格的位置、放置顺序的圆排列数，答案为

$n\sum_{i=n\bmod 2}^{\lfloor n/2\rfloor} 2\cdot{n-i\choose i}\cdot(n-i-1)!$

注意其中系数 n 是环旋转的方案数，系数 2 是 A,B 交错有两种的可能的排列。

代码

#include <bits/stdc++.h>
using ll = long long;
const int P = 1e9 + 7;
int power(int a, int b) {
    int res = 1;
    while (b) {
        if (b & 1)
            res = (ll)res * a % P;
        a = (ll)a * a % P;
        b >>= 1;
    }
    return res;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int n;
    std::cin >> n;
    std::vector<int> fac(n + 1), ifac(n + 1);
    fac[0] = ifac[0] = 1;
    for (int i = 1; i <= n; ++i)
        fac[i] = (ll)fac[i - 1] * i % P;
    ifac[n] = power(fac[n], P - 2);
    for (int i = n; i > 1; --i)
        ifac[i - 1] = (ll)ifac[i] * i % P;
    auto choose = [&](int x, int y) -> int {
        return (ll)fac[x] * ifac[y] % P * ifac[x - y] % P;
    };
    int ans = 0;
    for (int i = n % 2; i <= n / 2; i += 2)
        ans = (ans + (ll)choose(n - i, i) * 2 * fac[n - i - 1]) % P;
    ans = (ll)ans * n % P;
    std::cout << ans << '\n';
    return 0;
}
